3.1 What Is JavaScript, and Where Does It Run?

3.2 Variables: The Computer's Memory

A variable is a labelled box that stores a value. Modern JavaScript gives you two keywords:

let score = 0;          // let: a box whose contents CAN change
score = 10;             // fine
const APP_NAME = "JobFinder";  // const: contents can NEVER be reassigned
// APP_NAME = "Other";         // ERROR — and that's a good thing
// Rule of thumb professionals follow: use const by default,
// switch to let only when you know the value must change.

The Core Data Types

const jobTitle = "Front-End Developer";  // string  (text, in quotes)
const salary = 250000;                    // number
const isRemote = true;                    // boolean (true or false)
let assignedMentor;                       // undefined (declared, no value yet)
const middleName = null;                  // null (deliberately empty)
console.log(typeof jobTitle);  // "string" — typeof reveals any value's type

3.3 Operators: Doing Things With Values

// Arithmetic
console.log(10 + 3);   // 13
console.log(10 % 3);   // 1  — remainder, surprisingly useful
console.log(2 ** 5);   // 32 — exponent
// Strings: + joins, but template literals are better
const name = "Seyi";
console.log("Hello, " + name + "!");        // old way
console.log(`Hello, ${name}! You have ${3 * 4} new messages.`); // modern way
// Comparison — ALWAYS use === (checks value AND type)
console.log(5 === 5);    // true
console.log(5 === "5");  // false — number vs string
console.log(5 !== 4);    // true
// Logical
console.log(true && false); // AND — both must be true
console.log(true || false); // OR  — at least one true
console.log(!true);         // NOT — flips it

3.5 Loops: Doing Things Repeatedly

// The classic for loop: start; keep-going condition; step
for (let i = 1; i <= 5; i++) {
  console.log(`Lap number ${i}`);
}
// while loop: when you don't know how many repetitions in advance
let battery = 100;
while (battery > 0) {
  console.log(`Battery at ${battery}%`);
  battery -= 25;
}
// Looping over an array — the loop you'll use most in real work
const skills = ["HTML", "CSS", "JavaScript"];
for (const skill of skills) {
  console.log(`I am learning ${skill}`);
}

3.6 Functions: Reusable Machines

A function is a named, reusable block of code: ingredients go in (parameters), a result comes out (return value). Functions are how you stop repeating yourself.

// Classic declaration
function calculateVAT(amount) {
  return amount * 0.075;          // Nigeria's 7.5% VAT
}
// Arrow function — the modern style you'll see everywhere
const formatNaira = (amount) => `₦${amount.toLocaleString()}`;
const price = 120000;
const vat = calculateVAT(price);
console.log(`Price: ${formatNaira(price)}`);
console.log(`VAT:   ${formatNaira(vat)}`);
console.log(`Total: ${formatNaira(price + vat)}`);

3.7 Arrays: Ordered Lists of Anything

const jobs = ["Designer", "Developer", "Writer"];
console.log(jobs[0]);        // "Designer" — counting starts at 0
console.log(jobs.length);    // 3
jobs.push("Analyst");        // add to the end
jobs.pop();                  // remove from the end
// The power trio you will use DAILY as a front-end developer:
const prices = [1000, 2500, 4000, 800];
const doubled  = prices.map(p => p * 2);        // transform every item
const cheap    = prices.filter(p => p < 2000);  // keep items that pass a test
const total    = prices.reduce((sum, p) => sum + p, 0); // boil down to one value
console.log(doubled); // [2000, 5000, 8000, 1600]
console.log(cheap);   // [1000, 800]
console.log(total);   // 8300
WHY map AND filter MATTER SO MUCH Every list you will ever render in React (Part 7) — job cards, products, messages — is an array transformed with .map(). Master it now in the console and React will feel natural later.

3.9 Checkpoint Project 1: The Number-Guessing Game

FILErt3-js/game.js
const secret = Math.floor(Math.random() * 100) + 1; // 1 to 100
let attempts = 0;
let guess = null;
while (guess !== secret) {
  guess = Number(prompt("Guess the number (1-100):"));
  attempts++;
  if (Number.isNaN(guess)) {
    alert("That is not a number. Try again.");
  } else if (guess < secret) {
    alert("Too low!");
  } else if (guess > secret) {
    alert("Too high!");
  } else {
    alert(`Correct! You got it in ${attempts} attempts.`);
  }
}

3.10 Checkpoint Project 2: Console Expense Tracker

FILErt3-js/expenses.js
const expenses = [];
function addExpense(description, amount, category) {
  expenses.push({ description, amount, category, date: new Date() });
}
function getTotal() {
  return expenses.reduce((sum, e) => sum + e.amount, 0);
}
function getByCategory(category) {
  return expenses.filter(e => e.category === category);
}
function printReport() {
  console.log("=== EXPENSE REPORT ===");
  expenses.forEach(e =>
    console.log(`${e.description.padEnd(20)} ₦${e.amount} [${e.category}]`));
  console.log(`TOTAL: ₦${getTotal()}`);
}
// Test drive
addExpense("Data bundle", 5000, "internet");
addExpense("Transport", 3500, "travel");
addExpense("Lunch", 2000, "food");
addExpense("Hosting", 8000, "internet");
printReport();
console.log("Internet expenses:", getByCategory("internet"));
